The change reworks the retry path. Key points:

  • backoff is now exponential
  • maxRetries defaults to 3
async function retry(fn, n = 3) {
  for (let i = 0; i < n; i++) {
    try { return await fn(); }
    catch (e) { await wait(2 ** i * 100); }
  }
}
